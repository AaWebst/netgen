/* Simplified DPDK packet generator */
#include <rte_eal.h>
#include <rte_ethdev.h>
#include <rte_mbuf.h>

#define MAX_PKT_BURST 64

int main(int argc, char **argv) {
    struct rte_mempool *mbuf_pool;
    uint16_t portid = 0;
    
    // Initialize DPDK
    int ret = rte_eal_init(argc, argv);
    
    // Create memory pool
    mbuf_pool = rte_pktmbuf_pool_create("MBUF_POOL", 8192,
        250, 0, RTE_MBUF_DEFAULT_BUF_SIZE, rte_socket_id());
    
    // Configure port
    struct rte_eth_conf port_conf = {};
    rte_eth_dev_configure(portid, 1, 1, &port_conf);
    rte_eth_rx_queue_setup(portid, 0, 128, 
        rte_eth_dev_socket_id(portid), NULL, mbuf_pool);
    rte_eth_tx_queue_setup(portid, 0, 512,
        rte_eth_dev_socket_id(portid), NULL);
    rte_eth_dev_start(portid);
    
    // Main loop - send packets at line rate
    struct rte_mbuf *bufs[MAX_PKT_BURST];
    while (1) {
        // Allocate mbufs
        for (int i = 0; i < MAX_PKT_BURST; i++) {
            bufs[i] = rte_pktmbuf_alloc(mbuf_pool);
            // Build packet in bufs[i]->data
        }
        
        // Send burst
        uint16_t sent = rte_eth_tx_burst(portid, 0, bufs, MAX_PKT_BURST);
        
        // Free unsent
        for (int i = sent; i < MAX_PKT_BURST; i++) {
            rte_pktmbuf_free(bufs[i]);
        }
    }
    
    return 0;
}
